Unity转换Terrain为mesh的简单功能实现；

  1using UnityEngine;
  2using UnityEditor;
  3
  4public class TerrainToMeshWindow : EditorWindow
  5{
  6
  7    [MenuItem("Window/TerrainToMeshWindow")]
  8    static void Init()
  9    {
 10        TerrainToMeshWindow window = (TerrainToMeshWindow)EditorWindow.GetWindow(typeof(TerrainToMeshWindow));
 11        window.Show();
 12    }
 13    int row = 5;
 14    int column = 5;
 15    Terrain terrain = null;
 16
 17    void OnGUI()
 18    {
 19        EditorGUILayout.BeginVertical();
 20        terrain = EditorGUILayout.ObjectField("terrain", terrain, typeof(Terrain), true) as Terrain;
 21        row = EditorGUILayout.IntField("row", row);
 22        column = EditorGUILayout.IntField("column", column);
 23        if (GUILayout.Button("convert"))
 24        {
 25            if( row * column * 6 > 65535)
 26            {
 27                Debug.LogError("row or column is too big! index buffer can't use 16 bit");
 28                return;
 29            }
 30
 31            TerrainData terrainData = terrain.terrainData;
 32            int vertexCount = (row + 1) * (column + 1);
 33
 34            // vertex && uv
 35            Vector3[] vertices = new Vector3[vertexCount];
 36            Vector2[] uvs = new Vector2[vertexCount];
 37            for (int i = 0; i <= row; i++)
 38            {
 39                for (int j = 0; j <= column; j++)
 40                {
 41                    float width = terrainData.size.x;
 42                    float length = terrainData.size.z;
 43
 44                    float x = (float)i / row * width;
 45                    float z = (float)j / column * length;
 46                    float y = terrain.SampleHeight(new Vector3(x, 0, z));
 47
 48                    vertices[i * (column + 1) + j] = new Vector3(x, y, z);
 49
 50                    uvs[i * (column + 1) + j] = new Vector2((float)i / row, (float)j / column);
 51                }
 52            }
 53
 54            // index
 55            int indexCount = row * column * 6;
 56            int[] indices = new int[indexCount];
 57            for (int i = 0; i < row; i++)
 58            {
 59                for (int j = 0; j < column; j++)
 60                {
 61                    indices[(i * column + j) * 6] = i * (column + 1) + j;
 62                    indices[(i * column + j) * 6 + 1] = i * (column + 1) + j + 1;
 63                    indices[(i * column + j) * 6 + 2] = (i + 1) * (column + 1) + j;
 64
 65                    indices[(i * column + j) * 6 + 3] = i * (column + 1) + j + 1;
 66                    indices[(i * column + j) * 6 + 4] = (i + 1) * (column + 1) + j + 1;
 67                    indices[(i * column + j) * 6 + 5] = (i + 1) * (column + 1) + j;
 68                }
 69            }
 70
 71            // normal
 72            Vector3[] normals = new Vector3[vertexCount];
 73            int[] normalcounts = new int[vertexCount];
 74            for (int i = 0; i < vertexCount; i++)
 75            {
 76                normals[i] = Vector3.zero;
 77                normalcounts[i] = 0;
 78            }
 79            for (int i = 0; i < row; i++)
 80            {
 81                for (int j = 0; j < column; j++)
 82                {
 83                    int vid0 = i * (column + 1) + j;
 84                    int vid1 = i * (column + 1) + j + 1;
 85                    int vid2 = (i + 1) * (column + 1) + j;
 86                    int vid3 = (i + 1) * (column + 1) + j + 1;
 87
 88                    Vector3 normal0 =Vector3.Cross(vertices[vid2]-vertices[vid0], vertices[vid2]-vertices[vid0]).normalized;
 89
 90                    Vector3 normal1 =Vector3.Cross(vertices[vid3]-vertices[vid1], vertices[vid2]-vertices[vid1]).normalized;
 91
 92                    normals[vid0] += normal0;
 93                    normals[vid1] += normal0;
 94                    normals[vid2] += normal0;
 95                    normalcounts[vid0]++;
 96                    normalcounts[vid1]++;
 97                    normalcounts[vid2]++;
 98
 99                    normals[vid1] += normal1;
100                    normals[vid2] += normal1;
101                    normals[vid3] += normal1;
102                    normalcounts[vid1]++;
103                    normalcounts[vid2]++;
104                    normalcounts[vid3]++;
105                }
106            }
107            for (int i = 0; i < vertexCount; i++)
108            {
109                normals[i] /= normalcounts[i];
110                normals[i].Normalize();
111            }
112
113            Mesh mesh = new Mesh();
114            mesh.name = terrain.name;
115            mesh.vertices = vertices;
116            mesh.uv = uvs;
117            mesh.triangles = indices;
118            mesh.normals = normals;
119
120            AssetDatabase.CreateAsset(mesh, "Assets/test.asset");
121            AssetDatabase.Refresh();
122        }
123        EditorGUILayout.EndVertical();
124    }
125}
